<script>
let foods=JSON.parse(localStorage.foods||"[]");
let cart=[];
let wa=localStorage.wa||"919999999999";
let current=null,opti=0,qty=1;
let filterCat="All",searchTxt="";

function render(){
  foodGrid.innerHTML="";
  foods.filter(f=>
    (filterCat=="All"||f.cat==filterCat) &&
    f.name.toLowerCase().includes(searchTxt)
  ).forEach((f,i)=>{
    if(!f.active)return;
    foodGrid.innerHTML+=`
    <div class="card">
      <img src="${f.img}">
      <div class="info">
        <h3>${f.name}</h3>
        <div class="price">From ₹${f.options[0].price}</div>
      </div>
      <button class="add" onclick="openModal(${i})">+</button>
    </div>`;
  });
}
render();

function setCat(c,b){
  filterCat=c;
  document.querySelectorAll(".cat button").forEach(x=>x.classList.remove("active"));
  b.classList.add("active");
  render();
}
function searchFood(v){searchTxt=v.toLowerCase();render();}

function openModal(i){
  current=foods[i];opti=0;qty=1;
  mname.innerText=current.name;
  mopts.innerHTML=current.options.map((o,idx)=>
    `<div class="option">
      <input type="radio" name="o" ${idx==0?"checked":""} onclick="opti=${idx}">
      ${o.name} - ₹${o.price}
    </div>`).join("");
  q.innerText=qty;
  modal.classList.add("active");
}

function addToCart(){
  cart.push({...current,option:current.options[opti],qty});
  modal.classList.remove("active");
  cartBar.style.display="flex";
  cartCount.innerText=cart.length+" items";
}

function openPage(p,b){
  document.querySelectorAll(".page").forEach(x=>x.classList.remove("active"));
  document.getElementById(p).classList.add("active");
  document.querySelectorAll("nav button").forEach(x=>x.classList.remove("active"));
  if(b)b.classList.add("active");
  if(p=="cart")renderCart();
}

function renderCart(){
  let t=0;
  cartList.innerHTML="";
  cart.forEach((c,i)=>{
    let p=c.option.price*c.qty;t+=p;
    cartList.innerHTML+=`
    <div class="cartItem">
      ${c.name} (${c.option.name}) x ${c.qty} – ₹${p}
      <button onclick="cart.splice(${i},1);renderCart()">Remove</button>
    </div>`;
  });
  total.innerText="Total ₹"+t;
}

function orderWA(){
  let msg="Order from Fair Foods\n\n";
  cart.forEach(c=>{
    msg+=`${c.name} (${c.option.name}) x ${c.qty}\n`;
  });
  window.open(`https://wa.me/${wa}?text=${encodeURIComponent(msg)}`);
}

/* ADMIN */
function login(){if(pass.value=="admin123")panel.style.display="block";}
function addOpt(){opts.innerHTML+=`<input placeholder="Option"><input placeholder="Price">`;}
function addItem(){
  let ins=[...opts.querySelectorAll("input")],o=[];
  for(let i=0;i<ins.length;i+=2)o.push({name:ins[i].value,price:+ins[i+1].value});
  foods.push({name:iname.value,img:iimg.value,cat:icat.value,options:o,active:true});
  localStorage.foods=JSON.stringify(foods);
  opts.innerHTML="";render();renderAdmin();
}
function renderAdmin(){
  adminList.innerHTML="";
  foods.forEach((f,i)=>{
    adminList.innerHTML+=`
    <div class="adminCard">
      ${f.name}
      <button onclick="f.active=!f.active;localStorage.foods=JSON.stringify(foods);render()">ON/OFF</button>
      <button onclick="foods.splice(${i},1);localStorage.foods=JSON.stringify(foods);render();renderAdmin()">Delete</button>
    </div>`;
  });
}
function saveWA(){localStorage.wa=wno.value;alert("Saved")}
renderAdmin();
</script>
